Theory AbstractDomainModel

theory AbstractDomainModel
  imports Complex_Main Timed_Methods
begin

locale abstract_domain_model =
  fixes broadcaster :: "'p::finite"
    and faulty :: "'p set"
    and opt_quorum_member :: "'opQ ⇒ 'p ⇒ bool"
    and maj_quorum_member :: "'vtQ ⇒ 'p ⇒ bool"
    and classic_quorum_member :: "'quQ ⇒ 'p ⇒ bool"
    and amplification_quorum_member :: "'amQ ⇒ 'p ⇒ bool"
    and commit_quorum_member :: "'coQ ⇒ 'p ⇒ bool"
  assumes broadcaster_not_in_classic_quorum:
    "¬ classic_quorum_member quQ broadcaster"
    and broadcaster_not_in_maj_quorum:
    "¬ maj_quorum_member vtQ broadcaster"
    and broadcaster_not_in_opt_quorum:
    "¬ opt_quorum_member opQ broadcaster"
    and non_faulty_commit_quorum:
    "∃coQ. ∀p. commit_quorum_member coQ p ⟶ p ∉ faulty"
    and non_faulty_amplification_quorum:
    "∃amQ. ∀p. amplification_quorum_member amQ p ⟶ p ∉ faulty"
    and non_faulty_maj_quorum:
    "∃vtQ. ∀p. maj_quorum_member vtQ p ⟶ p ∉ faulty"
    and non_faulty_classic_quorum:
    "∃quQ. ∀p. classic_quorum_member quQ p ⟶ p ∉ faulty"
    and opt_quorum_has_nonfaulty_member:
    "∃p. p ∉ faulty ∧ opt_quorum_member opQ p"
    and opt_quorum_intersection:
    "broadcaster ∈ faulty ⟶ (∃p. p ∉ faulty ∧ opt_quorum_member opQ⇩1 p ∧ opt_quorum_member opQ⇩2 p)"
    and opt_maj_quorum_intersection:
    "broadcaster ∈ faulty ⟶ (∃p. p ∉ faulty ∧ opt_quorum_member opQ p ∧ maj_quorum_member vtQ p)"
    and opt_quorum_intersects_classic_quorum:
    "broadcaster ∈ faulty ⟶ (∃p. p ∉ faulty ∧ opt_quorum_member opQ p ∧ classic_quorum_member quQ p)"
    and classic_quorum_intersection:
    "broadcaster ∈ faulty ⟶ (∃p. p ∉ faulty ∧ classic_quorum_member quQ⇩1 p ∧ classic_quorum_member quQ⇩2 p)"
    and quorum_not_faulty:
    "∃p. p ∉ faulty ∧ classic_quorum_member Q p"
    and maj_quorum_has_nonfaulty_member:
    "∃p. p ∉ faulty ∧ maj_quorum_member vtQ p"
    and amplification_quorum_has_nonfaulty_member:
    "∃p. p ∉ faulty ∧ amplification_quorum_member amQ p"
    and opt_quorum_contains_maj_quorum:
    "broadcaster ∈ faulty ⟶ (∃vtQ. ∀p. maj_quorum_member vtQ p ⟶ p ∉ faulty ∧ opt_quorum_member opQ p)"
    and commit_quorum_contains_non_faulty_amplification_quorum:
    "∃amQ. ∀p. amplification_quorum_member amQ p ⟶ p ∉ faulty ∧ commit_quorum_member coQ p"

end
